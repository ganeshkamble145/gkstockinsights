function calculateCryptoScore(coin) {
  const scores = {};

  // ── Factor 1: Multibagger Potential (30%) ──
  // Based on ATH recovery multiple or analyst target
  const multiple = coin.ai_analyst.multibagger_estimate.return_multiple;
  scores.potential = multiple == null ? 50
    : multiple >= 10 ? 95
    : multiple >= 5  ? 80
    : multiple >= 3  ? 65
    : multiple >= 2  ? 50
    : 30;

  // ── Factor 2: Fundamental Quality (25%) ──
  const fundMap = { STRONG: 90, ADEQUATE: 65, WEAK: 30, UNCLEAR: 45 };
  scores.fundamentals = fundMap[coin.fundamentals.fundamental_verdict] ?? 45;

  // ── Factor 3: Technical Position (20%) ──
  // RSI-based: oversold = good entry, overbought = risky entry
  const rsi = coin.technical.rsi_14;
  scores.technical = rsi == null ? 50
    : rsi < 30 ? 95      // oversold — excellent entry zone
    : rsi < 40 ? 80      // approaching oversold — good entry
    : rsi < 60 ? 65      // healthy range
    : rsi < 70 ? 45      // getting expensive
    : 20;                // overbought — wait for pullback

  // ── Factor 4: Research Sentiment (15%) ──
  const sentMap = {
    BULLISH: 90, NEUTRAL: 55,
    BEARISH: 20, 'NO COVERAGE': 40
  };
  scores.sentiment = sentMap[coin.research_desk.sentiment] ?? 40;

  // ── Factor 5: Tokenomics Health (10%) ──
  const tokMap = { LOW: 90, MEDIUM: 60, HIGH: 25 };
  scores.tokenomics = tokMap[coin.fundamentals.tokenomics_risk] ?? 50;

  // ── Composite ──
  const composite = Math.round(
    (scores.potential    * 0.30) +
    (scores.fundamentals * 0.25) +
    (scores.technical    * 0.20) +
    (scores.sentiment    * 0.15) +
    (scores.tokenomics   * 0.10)
  );

  // ── Hard overrides ──
  // Verified price must be < ₹200
  if (coin.price_inr >= 200) {
    return { composite: 0, tier: TIERS.EXCLUDED,
             reason: "Price ≥ ₹200 — excluded from this tab" };
  }
  // No real utility → cap at HOLD regardless of score
  if (coin.fundamentals.real_world_utility === 'NONE') {
    return {
      composite: Math.min(composite, 45),
      tier: cryptoTier(Math.min(composite, 45)),
      reason: "No real utility — capped at HOLD"
    };
  }

  return {
    composite: Math.min(Math.max(composite, 0), 100),
    breakdown: scores,
    tier: cryptoTier(composite)
  };
}

function cryptoTier(score) {
  if (score >= 80) return { label: 'STRONG BUY',       badge: '🚀',  color: 'green'  };
  if (score >= 65) return { label: 'BUY / ACCUMULATE', badge: '✅',  color: 'green'  };
  if (score >= 50) return { label: 'SPECULATIVE BUY',  badge: '⚡',  color: 'blue'   };
  if (score >= 35) return { label: 'HOLD / WATCH',     badge: '⚠️', color: 'amber'  };
  return              { label: 'AVOID',                badge: '❌',  color: 'red'    };
}

PART 5 — COMPOSITE SCORING ENGINE
(Client-side JavaScript — run after AI analysis completes)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

function calculateCryptoScore(coin) {
  const scores = {};

  // ── Factor 1: Multibagger Potential (30%) ──
  // Based on ATH recovery multiple or analyst target
  const multiple = coin.ai_analyst.multibagger_estimate.return_multiple;
  scores.potential = multiple == null ? 50
    : multiple >= 10 ? 95
    : multiple >= 5  ? 80
    : multiple >= 3  ? 65
    : multiple >= 2  ? 50
    : 30;

  // ── Factor 2: Fundamental Quality (25%) ──
  const fundMap = { STRONG: 90, ADEQUATE: 65, WEAK: 30, UNCLEAR: 45 };
  scores.fundamentals = fundMap[coin.fundamentals.fundamental_verdict] ?? 45;

  // ── Factor 3: Technical Position (20%) ──
  // RSI-based: oversold = good entry, overbought = risky entry
  const rsi = coin.technical.rsi_14;
  scores.technical = rsi == null ? 50
    : rsi < 30 ? 95      // oversold — excellent entry zone
    : rsi < 40 ? 80      // approaching oversold — good entry
    : rsi < 60 ? 65      // healthy range
    : rsi < 70 ? 45      // getting expensive
    : 20;                // overbought — wait for pullback

  // ── Factor 4: Research Sentiment (15%) ──
  const sentMap = {
    BULLISH: 90, NEUTRAL: 55,
    BEARISH: 20, 'NO COVERAGE': 40
  };
  scores.sentiment = sentMap[coin.research_desk.sentiment] ?? 40;

  // ── Factor 5: Tokenomics Health (10%) ──
  const tokMap = { LOW: 90, MEDIUM: 60, HIGH: 25 };
  scores.tokenomics = tokMap[coin.fundamentals.tokenomics_risk] ?? 50;

  // ── Composite ──
  const composite = Math.round(
    (scores.potential    * 0.30) +
    (scores.fundamentals * 0.25) +
    (scores.technical    * 0.20) +
    (scores.sentiment    * 0.15) +
    (scores.tokenomics   * 0.10)
  );

  // ── Hard overrides ──
  // Verified price must be < ₹200
  if (coin.price_inr >= 200) {
    return { composite: 0, tier: TIERS.EXCLUDED,
             reason: "Price ≥ ₹200 — excluded from this tab" };
  }
  // No real utility → cap at HOLD regardless of score
  if (coin.fundamentals.real_world_utility === 'NONE') {
    return {
      composite: Math.min(composite, 45),
      tier: cryptoTier(Math.min(composite, 45)),
      reason: "No real utility — capped at HOLD"
    };
  }

  return {
    composite: Math.min(Math.max(composite, 0), 100),
    breakdown: scores,
    tier: cryptoTier(composite)
  };
}

function cryptoTier(score) {
  if (score >= 80) return { label: 'STRONG BUY',       badge: '🚀',  color: 'green'  };
  if (score >= 65) return { label: 'BUY / ACCUMULATE', badge: '✅',  color: 'green'  };
  if (score >= 50) return { label: 'SPECULATIVE BUY',  badge: '⚡',  color: 'blue'   };
  if (score >= 35) return { label: 'HOLD / WATCH',     badge: '⚠️', color: 'amber'  };
  return              { label: 'AVOID',                badge: '❌',  color: 'red'    };
}

TAX CALCULATOR (client-side JS):
function calculatePostTaxReturn(coin, investmentInr) {
  const buyPrice    = coin.price_inr;
  const targetPrice = coin.ai_analyst.multibagger_estimate.target_price_inr;

  if (!buyPrice || !targetPrice || targetPrice <= buyPrice) {
    return { postTaxReturn: null, note: "No valid target price available" };
  }

  const units         = investmentInr / buyPrice;
  const grossValue    = units * targetPrice;
  const grossGain     = grossValue - investmentInr;
  const tdsDeducted   = grossValue * 0.01;
  const taxOnGain     = grossGain * 0.30;
  const netAfterTax   = grossValue - taxOnGain;
  const postTaxMultiple = netAfterTax / investmentInr;
  const years         = coin.ai_analyst.multibagger_estimate.horizon_years || 2;
  const postTaxCAGR   = (Math.pow(postTaxMultiple, 1/years) - 1) * 100;

  return {
    investment_inr:     investmentInr,
    gross_value_inr:    Math.round(grossValue),
    gross_gain_inr:     Math.round(grossGain),
    gross_multiple:     (grossValue / investmentInr).toFixed(2),
    tds_deducted_inr:   Math.round(tdsDeducted),
    tax_on_gain_inr:    Math.round(taxOnGain),
    net_value_inr:      Math.round(netAfterTax),
    post_tax_multiple:  postTaxMultiple.toFixed(2),
    post_tax_cagr_pct:  postTaxCAGR.toFixed(1),
    horizon_years:      years,
    tax_note: `30% tax = ₹${Math.round(taxOnGain).toLocaleString('en-IN')}
               on ₹${Math.round(grossGain).toLocaleString('en-IN')} gain.
               1% TDS = ₹${Math.round(tdsDeducted).toLocaleString('en-IN')}
               deducted at sale (claimable at ITR filing).
               Real return after tax: ${postTaxMultiple.toFixed(2)}x
               vs gross ${(grossValue/investmentInr).toFixed(2)}x`
  };
}

ON-CHAIN BONUS SCORING (applied on top of composite score):
function onChainBonus(coin) {
  let bonus = 0;
  const oc = coin.fundamentals;

  if (oc.developer_activity === 'HIGH')     bonus += 8;
  else if (oc.developer_activity === 'MEDIUM') bonus += 4;
  else if (oc.developer_activity === 'LOW')  bonus -= 5;

  if (oc.holder_trend === 'GROWING')        bonus += 7;
  else if (oc.holder_trend === 'DECLINING') bonus -= 8;

  if (oc.real_world_utility === 'HIGH')     bonus += 5;
  else if (oc.real_world_utility === 'NONE') bonus -= 10;

  if (oc.upcoming_catalysts?.length > 0)   bonus += 5;

  return Math.max(-15, Math.min(15, bonus)); // cap at ±15
}
finalScore = Math.min(100, Math.max(0, compositeScore + onChainBonus(coin)));
