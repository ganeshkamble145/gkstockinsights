  2. Batch fetching — max 5 stocks per batch, 600ms delay:
     async function fetchInBatches(tickers, batchSize=5, delay=600) {
       const results = [];
       for (let i = 0; i < tickers.length; i += batchSize) {
         const batch = tickers.slice(i, i + batchSize);
         const res = await Promise.allSettled(
           batch.map(t => fetchStockData(formatTicker(t)))
         );
         results.push(...res);
         if (i + batchSize < tickers.length)
           await new Promise(r => setTimeout(r, delay));
       }
       return results;
     }

  3. 10-second timeout per fetch:
     async function fetchStockData(ticker) {
       const ctrl = new AbortController();
       const t = setTimeout(() => ctrl.abort(), 10000);
       try {
         const r = await fetch(`/api/public/yahoo-proxy?symbol=${ticker}`,
           { signal: ctrl.signal });
         clearTimeout(t);
         return await r.json();
       } catch { clearTimeout(t); return { symbol: ticker, error: true }; }
     }

  4. Safe AI JSON parser:
     function safeParseAI(text) {
       try {
         const c = text.replace(/```json|```/gi,'').trim();
         return JSON.parse(c);
       } catch {
         const a = text.match(/\[[\s\S]*\]/);
         if (a) try { return JSON.parse(a[0]); } catch {}
         const o = text.match(/\{[\s\S]*\}/);
         if (o) try { return [JSON.parse(o[0])]; } catch {}
         return null;
       }
     }

     function sanitizeStock(s) {

         composite_score: Math.min(100, Math.max(0,

       };
     }

function calculateScore(stock, tab) {
  const { price_5d_change_pct: mom, volume_ratio: vol,
          rsi, pe_ratio: pe } = stock;

  // Momentum (30%)
  const m = Math.min(Math.max((mom + 10) / 20 * 100, 0), 100);

  // Volume (25%)
  const v = vol > 0 ? Math.min(vol / 3 * 100, 100) : 50;

  // RSI (25%) — neutral 50 if unavailable
  const r = !rsi ? 50
    : rsi >= 40 && rsi <= 65 ? 100
    : (rsi >= 30 && rsi < 40) || (rsi > 65 && rsi <= 75) ? 60 : 20;

  // P/E (20%) — different thresholds for large-cap vs penny
  const thresholds = tab === 'nifty'
    ? [20, 30, 50]   // large-cap benchmarks
    : [15, 25, 40];  // penny/small-cap benchmarks
  const p = !pe || pe <= 0 ? 50
    : pe < thresholds[0] ? 90
    : pe < thresholds[1] ? 70
    : pe < thresholds[2] ? 50 : 20;

  return Math.round(Math.min(Math.max(
    m * 0.30 + v * 0.25 + r * 0.25 + p * 0.20, 0), 100));
}

Score → Recommendation tier:
  ≥ 75 → STRONG BUY ⭐ | 60–74 → BUY ✅
  45–59 → HOLD ⚠️ | 25–44 → AVOID 🔻 | < 25 → SELL ❌

  function getCache(key, ttl) {
    try {
      const x = JSON.parse(localStorage.getItem(key));
      if (!x || Date.now() - x.ts > ttl * 60000) return null;
      return x.data;
    } catch { return null; }
  }

    localStorage.setItem(key, JSON.stringify({ data, ts: Date.now() }));
